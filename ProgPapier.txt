const float c_MinXSlotDisplay = -6.5f;
const float c_MaxXSlotDisplay = 6.5f;

void Start() {
	// Stuffs
	
	PopulateSlots();
	
	// Stuffs
}

void OnMouseDown() {
	for (int i =0; i < m_ToolSlots; ++i) {
		Vector3 destination = m_ToolSlots[i].transform.position;
		destination.x -= 1.6f;
		
		// Must be tested !!
		m_ToolSlots[i].transform.position.Lerp(from: m_ToolSlots[i].transform.position,
			to: destination, Time.deltaTime);
	}
	
	DetermineSlotsDisplay();
}

void DetermineSlotsDisplay() {
	for (int i =0; i < m_ToolSlots; ++i) {
		float currentXPosition = m_ToolSlots[i].transform.position.x;
		m_ToolSlots[i].SetActive(currentXPosition > c_MinXSlotDisplay
								 && currentXPosition < c_MaxXSlotDisplay);
	}
}

void PopulateSlots() {
	m_ToolSlots = new List<GameObject>();
	GameObject toolsInitParent = GameObject.Find("toolsInitParent");
	
	for (int i = 0; i < toolsInitParent.childCount; ++i) {
		m_ToolSlots.Add(toolsInitParent.transform.GetChild(i).gameobject);
	}

	float computedXPosition = c_MinXSlotDisplay - 1.6f;
	for (int i =0; i < m_ToolSlots; ++i) {
		computedXPosition += 1.6f;
		Vector3 newPosition = new Vector3(computedXPosition, -4.6f, 0);
		m_ToolSlots[i].position = newPosition;
	}
	
	DetermineSlotsDisplay();
}
